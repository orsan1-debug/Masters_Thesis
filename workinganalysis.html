# Keeps ALL estimators; subsetting happens in the figure calls, never here.
# `by` is a parameter so sections with different design columns (e.g. no `overlap`)
# can drop them: summarise_sim(df, by = c("n","p","outcome","estimator")).

summarise_sim <- function(data,
                          by = c("n","p","misspec","covcor", "outcome","overlap","estimator")) {
  data |>
    group_by(across(all_of(by))) |>
    summarise(
      m          = dplyr::n(),
      bias       = mean(tau_hat),
      empse      = sd(tau_hat),
      rmse       = sqrt(mean(tau_hat^2)),
      mcse_bias  = empse / sqrt(m),
      mcse_empse = empse / sqrt(2 * (m - 1)),
      mcse_rmse  = sd(tau_hat^2) / (2 * rmse * sqrt(m)),
      n_extreme  = sum(abs(tau_hat) > 10),
      .groups    = "drop"
    )
}

# ── 4. Core line plot: metric vs x, coloured by estimator ──
# y/ylab + x + ests are arguments; x-scale and facets are appended by the caller.
line_plot <- function(data, y, ylab, x = n, ests = est_main, from_zero = TRUE) {
  ests <- intersect(names(lab_est), ests)                 # canonical order, no ghost keys
  p <- data |>
    filter(estimator %in% ests) |>
    ggplot(aes({{ x }}, {{ y }}, colour = estimator, group = estimator)) +
    geom_line(linewidth = 0.7) + geom_point(size = 1.8) +
    scale_colour_manual(values = pal_est, breaks = ests,
                        labels = \(b) unname(lab_est[b])) +
    labs(y = ylab, colour = NULL) + theme_sim
  if (from_zero) p + expand_limits(y = 0) else p
}

# ── 5. Twin panel: RMSE | |Bias|, one shared legend ──
panel <- function(data, oc, x = n, ests = est_main, extra = NULL,
                  title = oc_labs[oc], from_zero = TRUE, compact = FALSE) {
  d <- filter(data, outcome == oc)
  drop0 <- \(x) sub("^(-?)0\\.", "\\1.", x)
  yl <- if (compact) NULL else "RMSE"
  yr <- if (compact) NULL else "|Bias|"
  l <- line_plot(d, rmse,      yl, x = {{ x }}, ests = ests, from_zero = from_zero) + extra +
    theme(legend.position = "none")
  r <- line_plot(d, abs(bias), yr, x = {{ x }}, ests = ests, from_zero = from_zero) + extra +
    theme(legend.position = "none")
  if (compact) {
    l <- l + ggtitle("RMSE") + scale_y_continuous(labels = drop0)
    r <- r + ggtitle("|Bias|") + scale_y_continuous(labels = drop0)
  }
  (l | r) + plot_layout(guides = "collect") +
    plot_annotation(title = title) & theme(legend.position = "bottom")
}

# ── 6. MSE decomposition: Bias² / Variance share, stacked area ──
pal_comp <- c("Variance" = "#377eb8", "Bias\u00b2" = "#e41a1c")

decomp_plot <- function(data, ests = est_main, x = n, cols = outcome,
                        x_scale = scale_x_n, xlab = "n",
                        col_lab = labeller(outcome = oc_labs, n = lab_n),
                        title = NULL) {

    pivot_longer(c(variance, bias_sq), names_to = "component", values_to = "share") |>
    mutate(component = factor(component, c("variance","bias_sq"),
                              c("Variance","Bias\u00b2"))) |>

    geom_area() +

               labeller = col_lab) +
    x_scale +
    scale_y_continuous(breaks = c(1), labels = scales::percent) +
    scale_fill_manual(values = pal_comp) +
    labs(x = xlab, y = "Share of MSE", fill = NULL, title = title) +
    theme_sim +
    theme(strip.text.y.left = element_text(angle = 0))
}

# ── 7. Performance table: estimators × design columns, ordered by mean metric ──
# cols="n" → flat.  cols=c("overlap","n") → c-spanners.
# group="outcome" → stacks all outcomes in one table with row groups.
perf_table <- function(data, ests = est_main, cols = "n",
                       top = "rmse", bottom = "abs_bias",
                       digits = 2, order_by = "rmse",
                       lab = lab_est, n_lab_tbl = n_lab,
                       span_lab = \(x) paste0("c = ", x), span_desc = TRUE,
                       title = "", subtitle = NULL,
                       group = NULL, group_lab = oc_labs) {
  stopifnot(length(cols) %in% 1:2,
            c(top, bottom, order_by) %in% c("rmse","bias","abs_bias","empse"))
  keep <- intersect(names(lab), ests)
  cvar <- cols[length(cols)]
  svar <- if (length(cols) == 2L) cols[[1]] else NULL
  pm   <- c(rmse="RMSE", bias="Bias", abs_bias="|Bias|", empse="EmpSE")
  f    <- paste0("%.", digits, "f")
  grp  <- !is.null(group)
  
  d <- data |> filter(estimator %in% keep) |> mutate(abs_bias = abs(bias)) |> mutate(
    Estimator = factor(unname(lab[estimator]), levels = unname(lab[keep])),
    cell = sprintf(paste0(f, "<br><small>(", f, ")</small>"), .data[[top]], .data[[bottom]]),
    avg  = ave(.data[[order_by]], estimator, FUN = mean),
    .col = if (is.null(svar)) paste0(cvar, "=", .data[[cvar]])
    else paste0(svar, "=", .data[[svar]], "|", cvar, "=", .data[[cvar]]))
  if (grp) d <- mutate(d, Group = factor(group_lab[.data[[group]]], levels = group_lab))
  
  cv  <- sort(unique(d[[cvar]]))
  sv  <- if (is.null(svar)) NULL else sort(unique(d[[svar]]), decreasing = span_desc)
  key <- if (is.null(svar)) paste0(cvar, "=", cv)
  else c(t(outer(sv, cv, \(s, c) paste0(svar, "=", s, "|", cvar, "=", c))))
  
  id_vars <- c(if (grp) "Group", "Estimator", ".col", "cell", "avg")
  wide <- d |> distinct(across(all_of(id_vars))) |>
    pivot_wider(names_from = .col, values_from = cell)
  wide <- if (grp) arrange(wide, Group, avg) else arrange(wide, avg)
  present <- intersect(key, names(wide))
  wide <- wide |> select(any_of("Group"), Estimator, all_of(present))
  
  inner_lab <- \(x) { x <- sub("^[^=]*=", "", sub(".*\\|", "", x))
  if (is.null(n_lab_tbl)) x else unname(ifelse(x %in% names(n_lab_tbl), n_lab_tbl[x], x)) }
  
  g <- if (grp) gt(wide, groupname_col = "Group") else gt(wide)
  g <- g |> tab_header(title = title,
                       subtitle = md(subtitle %||% paste0(pm[[top]], " (", pm[[bottom]], ")")))
  for (s in sv) g <- tab_spanner(g, span_lab(s), starts_with(paste0(svar, "=", s, "|")))
  g |>
    cols_label_with(all_of(present), inner_lab) |>
    fmt_markdown(all_of(present)) |> cols_align("center", all_of(present)) |>
    tab_style(cell_text(weight = "bold"), cells_body(Estimator)) |>
    opt_table_lines("default") |>
    tab_source_note(md(sprintf("_Top %s / bottom %s; sorted by mean %s._",
                               pm[[top]], pm[[bottom]], pm[[order_by]])))
}


# ── 8. Convergence / path-truncation table ──
# % reps where ALL selected estimators coincide (range < tol).
# k=2: bal0 ≡ bal05.  k≥3: full-path collapse.  Raw replicates, not summaries.
converge_table <- function(data, ests, by = c("overlap","n"),
                           cols = by[length(by)], tol = 1e-9,
                           outcome_name = NULL, overlap_vals = NULL,
                           n_lab_tbl = n_lab, title = NULL) {
  stopifnot(length(ests) >= 2, length(cols) == 1, cols %in% by)
  d <- filter(data, estimator %in% ests,
              if (is.null(outcome_name)) TRUE else outcome == outcome_name,
              if (is.null(overlap_vals)) TRUE else overlap %in% overlap_vals)
  
  wide <- d |>
    group_by(across(all_of(c(by, "sim")))) |>
    summarise(hit = (max(tau_hat) - min(tau_hat)) < tol, .groups = "drop") |>
    group_by(across(all_of(by))) |>
    summarise(pct = sprintf("%.1f%%", mean(hit) * 100), .groups = "drop") |>
    pivot_wider(names_from = all_of(cols), values_from = pct, names_prefix = paste0(cols, "="))
  
  ord  <- grep(paste0("^", cols, "="), names(wide), value = TRUE)
  ord  <- ord[order(as.numeric(sub(".*=", "", ord)))]
  wide <- select(wide, all_of(setdiff(by, cols)), all_of(ord))
  if ("overlap" %in% names(wide)) names(wide)[names(wide) == "overlap"] <- "c"
  
  g <- gt(wide)
  if (!is.null(title)) g <- tab_header(g, title = md(title))
  g |>
    cols_label_with(all_of(ord), \(x) { v <- sub(".*=", "", x)
    if (is.null(n_lab_tbl)) v else unname(ifelse(v %in% names(n_lab_tbl), n_lab_tbl[v], v)) }) |>
    cols_align("center")
}

# ── 9. Misspecification helpers ──
# ΔRMSE and Δ|Bias| = misspecified − correct at matching (n, outcome, overlap, estimator).
delta_ms <- function(summ, ests = est_main) {
  summ |>
    filter(estimator %in% ests) |>
    mutate(absbias = abs(bias)) |>
    select(n, outcome, overlap, estimator, misspec, rmse, absbias) |>
    pivot_wider(names_from = misspec, values_from = c(rmse, absbias)) |>
    mutate(d_rmse = rmse_TRUE - rmse_FALSE,
           d_bias = absbias_TRUE - absbias_FALSE)
}

delta_rmse <- function(summ, ests = est_main)
  delta_ms(summ, ests) |> transmute(n, outcome, overlap, estimator, delta = d_rmse)

Expected Results

E1 (Correct specification, strong overlap): Both methods should produce similar bias and RMSE. Any divergence here signals an implementation or DGP issue rather than a substantive finding. This serves as a sanity check for the simulation infrastructure [@keele2025; @tan2020b].

E2 (Overlap degradation): As the overlap scaler \(c\) decreases, bias and RMSE for cv.glmnet should increase faster than for cv.balnet. The performance gap should widen as overlap weakens [@keele2025; @chattopadhyay2020].

E3 (Misspecification): Under misspecification via Kang–Schafer transformations, cv.balnet should show a clear advantage in both bias and RMSE, exceeding the gap observed under correct specification. The interaction of misspecification with weak overlap should produce the largest relative advantage. This follows from the calibration loss controlling MSRE, which bounds IPW bias via Cauchy–Schwarz, whereas the MLE loss controls KL divergence, which does not directly bound IPW bias [@tan2020b, Prop. 3].

E4 (High dimensionality): With \(s\) fixed, increasing \(p\) should hurt both methods, but cv.balnet should be more robust. Under misspecification at high \(p\), RCAL selects sparser models than RML while achieving equal or better IPW estimation [@tan2020b].

E5 (Soft sparsity): As \(s\) increases with harmonic decay coefficients, the boundary between active and inactive covariates blurs under the AR(1) covariance structure. The balancing loss should handle this more robustly because it optimises for balance rather than prediction, retaining weakly-correlated covariates that happen to improve balance. There is no direct precedent in the project literature for this experiment; it is the simulation’s novel contribution.

E7 (Lambda interpretation and fixed-lambda behaviour): Fixed \(\lambda = 0.05\) should produce tighter balance but higher variance (lower effective sample size) than \(\lambda = 0.10\). The CV-selected \(\lambda\) should generally fall between these. Under weak overlap, fixed \(\lambda = 0.05\) may become infeasible due to path truncation, which is itself a diagnostic for positivity violations [@sverdrup2026].

E8 (Computational cost): cv.balnet must fit two separate propensity score models for ATE estimation (one for each potential outcome mean), roughly doubling computation relative to cv.glmnet’s single model [@sverdrup2026, Remark 1].

E9 (Variable selection): cv.balnet should consistently select fewer nonzero coefficients than cv.glmnet, especially under misspecification. This is consistent with the calibration loss requiring fewer covariates to achieve balance than the MLE loss requires for predictive accuracy [@tan2020b].

DGP 1

Data-Generating Process

We draw \(n\) observations of \(p\) covariates \(X_i = (X_{i1}, \dots, X_{ip})^\top \sim \mathcal{N}(0, \Sigma)\), where \(\Sigma_{jk} = \rho^{|j-k|}\) with \(\rho = 0\) (independent) or \(\rho = 0.5\) (AR(1)). The true propensity score is

\[e^*(X_i) = \left\{1 + \exp\left(\frac{X_{i1} - 0.5\,X_{i2} + 0.25\,X_{i3} + 0.1\,X_{i4}}{c}\right)\right\}^{-1}\]

where \(c > 0\) is an overlap scaler [@keele2025], and treatment is assigned as \(W_i \sim \text{Bernoulli}(e^*(X_i))\). Only the first four of \(p\) covariates are active in the propensity score, creating a sparse signal setting consistent with @tan2020.

Under the sharp null \(\tau = 0\), potential outcomes are generated as

\[\begin{aligned} Y_i^{\text{lin}} &= X_{i1} + 0.5\,X_{i2} + 0.5\,X_{i3} + 0.5\,X_{i4} + \varepsilon_i \\ Y_i^{\text{quad}} &= \sum_{j=1}^{4} \{\max(X_{ij}, 0)\}^2 + \varepsilon_i \\ Y_i^{\text{exp}} &= \sum_{j=1}^{4} \exp(X_{ij}/2) + \varepsilon_i \end{aligned}\]

where \(\varepsilon_i \sim \mathcal{N}(0, 1)\). Outcome specifications follow from @tan2020.

To introduce misspecification, the analyst observes transformed covariates \(\tilde{X}\) instead of \(X\) [@kangschafer2007]:

\[\begin{aligned} \tilde{X}_{i1} &= \exp(X_{i1}/2) \\ \tilde{X}_{i2} &= \frac{X_{i2}}{1 + \exp(X_{i1})} + 10 \\ \tilde{X}_{i3} &= (0.04\,X_{i1}\,X_{i3} + 0.6)^3 \\ \tilde{X}_{i4} &= (X_{i2} + X_{i4} + 20)^2 \end{aligned}\]

where \(\tilde{X}_{i1}, \dots, \tilde{X}_{i4}\) are subsequently standardized to have mean zero and unit variance, following @tan2020. For \(j = 5, \dots, p\), \(\tilde{X}_{ij} = X_{ij}\). Both logistic and linear regressions on \(\tilde{X}\) appear adequate by standard diagnostics, yet are misspecified since the true models are linear in \(X\) on the logit scale.

The simulation varies sample size \(n \in \{500, 1000, 5000\}\), covariate dimension \(p \in \{10, 50, 100, \dots\}\), and overlap \(c \in \{2, 1, 0.75, 0.5, 0.25\}\), with each experiment isolating one factor at a time.

As expected, misspecification and overlap compound, and MLE methods are considerably outperformed by balancing methods. MLE RMSE increases substantially, while the balancing estimators remain mostly flat.

Under misspecification with imperfect overlap, balnet’s attenuation bias is the cost of stability: RMSE stays bounded and improves with n, while MLE diverges significantly.

between methods, balancing protects against MLE’s divergence. Within balnet, λ is a bias–variance dial where less is better except in situations with small n and weak overlap, where cv.balnet shines. In large samples, its bias floor is too high.

Between Methods (MLE vs Balancing)

  • RMSE: Balancing consistently outperforms MLE at degraded overlap.
    • For MLE, increases in \(n\) and decreases in \(c\) degrade RMSE across all outcomes; for balancing, increases in \(n\) reduce RMSE.
    • Strong overlap shields MLE from misspecification. At \(c = 2, n \geq 1000\), MLE only loses to bal0 under linear and exponential outcomes, at quadratic it loses to all (potentially intestesting to see at higher C).
    • Balancing RMSE gradients across \(c\) are smooth and stable across \(n\) (intercept shifts down; effect varies by outcome complexity). MLE gradients are jagged and inconsistent across \(n\).
    • Divergence between methods is clearest under quadratic, then exponential outcomes.
    • Under linear outcomes, the balancing slope across \(c\) decreases noticeably with \(n\), and intercept reductions are smaller.
  • BIAS: MLE is generally less biased than balancing when \(n \leq 5000\), with exceptions:
    • Linear: at \(n = 500\), bal0 and bal05 less biased at \(c = 0.25\). At \(n = 5000\), bal0 less biased at \(c = 0.50\) (not \(c = 0.25\)).
    • Quadratic: at \(n = 1000\), MLE only less biased at \(c \leq 0.25\); all balancing less biased at \(c \geq 0.50\). At \(n = 5000\), MLE more biased than all balancing at all degraded overlap.
    • Exponential: at \(n = 5000\), MLE more biased than all balancing at all degraded overlap.
    • Motivation for AIPW to correct balancing estimators’ native bias.

  • Decomp: RMSE for balancing estimators are overwhelmingly more bias driven than MLE estimators. However, RMSE decompositions are effected heavily by outcome complexity.

    • For balancing estimators, RMSE becomes increasingly variance-driven as linear > exp > quadratic.

    • For MLE, this ordering holds at small \(n\) but reverses at \(n=5000\) when it begins to converge to the wrong \(\tau\) (quadratic most bias-driven, linear least).

    • MLE is mostly variance-driven at \(c < 2\) from \(n \geq 1000\); at \(c=2\) predominantly bias-driven for linear.

  • MLE produces 51 extreme IPW estimates (concentrated around quadratic outcomes and weak overlap, likely due to inversion), balancing methods produce 0. Where an extreme estimate \(|\hat{\tau}| > 10\) (no need for ad hoc weight trimming or truncation for balnet)

  • Effect of mispeccification compounds at lower overlap levels for both methods, but the interaction is substantially worse for MLE. (delta = RMSEmispec - RMSEcorrect)

    • MLE delta increases substantially as overlap weakens. (max at delta = 2.45 for quadratic outcome).

      • interaction effect grows with n for MLE, weak overlap accelerates K&S divergence . (eg: quad1 c=0.5: delta = 0.50 → 1.38 → 2.45 across n=500/1000/5000).

      • with strong overlap (c=2) divergence is mostly suppressed when mispeccified for linear and exponential outcomes.

      • compounding is clear for quadratic and linear outcomes; for exponential direction and ranking are clear, but the interaction is within MC noise (MCSE ≈ 0.13). Morris, White & Crowther (2019, Statistics in Medicine)

    • balancing estimators stay bounded, and flat near 0. (max at delta = 0.34 with linear outcome). the small interaction that exists is balnets misspec residual bias growing as overlap weakens.

      • Decomposing RMSE² into bias² + variance: misspecification shifts SD by at most 0.025 for bal0/bal05 — the penalty is almost entirely bias. balnetcv is the exception: SD falls under misspec as CV selects heavier λ, trading variance for bias.

  • Within balnet, less regularization wins: bal0 is best in 36/45 cells; CV’s λ is optimal where variance dominates.

    • balcv wins 6/45 cells: weak overlap and low n for non linear outcomes (c≤0.5, n≤1000) These settings make CV’s heavy shrinkage the right bias-variance trade. Otherwise, CV imposes an unnecessary bias floor.

    • Path truncation for bal0 and bal05. balnet0 = balnet05 exactly in 99.5% of reps at c=0.25/n=500 and 76.5% at n=1000. the path truncates above λ=0.05, as overlap is too weak to support that level of balance.

As expected, balancing estimators (using the balnet package) are more robust to high-dimensionality than MLE estimation. Bal0 (\(\lambda_{min} = 0\)) is still the best performing estimator.

Under high dimensionality, Bal05 performs similarly to Bal0, especially in low sample sizes. Increases in sample size increase the edge bal0 has over bal05 and other balancing estimators.

The stability of bal10, and that bal0 and bal05 converge toward it, suggest that the feasible \(\lambda_{endpoint}\) converges to \(\lambda \leq 0.1\) as \(p/n\) increases. Balnet.cv consistently selects \(\lambda\) higher than this, and performs similarly to the normalized MLE estimator (glmnet.cv).

These results support that the balnet.cv cross validation criterion (out-of-sample covariate balancing loss) is suboptimal.

1: Bal0 is still the strongest estimator*,

  • loses its edge in lower samples, where \(n=500\) bal0 converges with bal05 by \(p/n =.5\) , and \(p/n = 1\) when \(n=1000\).

    • these are the two estimators the with the least bias dominated RMSE.

      • bal05 consistently beats glm.

      • for all estimators, RMSE becomes more bias driven as \(p/n\) increases. glmnet/bal10 are the most bias dominated estimators, and start off almost completely bias dominated in linear specifications.

  • as \(p/n\) increases bal0 loses its efficiency against the normalized oracle as regularisation binds bal0’s allowed balance level, \(\lambda\), higher. efficiency result survives for quadratic specifications.

    • now loses to the normalized oracle under linear specifications when \(p/n \ge 1\). it also loses to the normalized oracle under exponential specifications at \(n=1000\) and \(p/n = 2\), at \(p/n = 1\) they approximately tie.

2: bal10 has stable variance and bias across all \(p/n\) levels.

  • bal10’s bias is stable across \(p/n\) as KKT conditions bound the maximum covariate imbalance to \(\lambda=.1\) (Sverdrup and Hastie 2026, eq. 10).

  • bal10’s flatness and that bal0 and bal05 converge towards it as \(p/n\) increases, indicating the feasible \(\lambda_{endpoint}\) converges with \(p/n\) increases to somewhere close (but below) \(\lambda = .1\) (can log regularisation path in future to determine where).

    • for all \(p/n ≥ 1\) the \(\lambda_{endpoint}\) exceeds 0.05, so bal05 and bal0 produce identical estimates.

3: at \(p/n \ge 0.5\) Balnet.cv and MLE and are consistently the worst performers. their comparitve performance depends on outcome complexity.

  • balcv degrades faster than glmnet as p grows and crosses below glm_hj . The loss function survives dimensionality, but the CV rule doesn’t. strong evidence against balnet.cv.

    • at \(p/n\) = 0.1, balcv still outperforms bal10 and glm.

      • CV still adapts at p/n = 0.1 but detaches by p/n = 0.5. FUTURE: logging λ_cv across the path will map the transition.

  • Under high dimensionality, once \(p/n = 1\) MLE outperforms balnet.cv decicivly for linear and moderatly for quadratic outcomes, approximatley tied under exponential outcomes.

  • increases in \(n\) reduce their variance, causing them to converge towards bal10.

4: larger samples reduces* RMSE for all estimators across all \(p/n\)

  • *for bal10, variance reduces with \(n\) while bias slightly increases. RMSE is also flat across the linear specification where RMSE is constantly \(\approx\) 0.25 .

5: outcome complexity effects bias/RMSE differently.

  • exponential: reduces bias/RMSE for all estimators (the one exception is bal0 at \(p/n =\) .1) , ranking remains the same.

  • quadratic: increases RMSE at low \(p/n\), but reduces it for the most biased estimators (balcv, glm) at high \(\mathbf{p}/\mathbf{n}\), where the bias reduction outweighs the higher outcome variance.

    • at \(n=1000\) bal.cv equivelent to glm.cv but loses to bal10 at \(p/n = .5\)
